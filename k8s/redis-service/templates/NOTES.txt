Thank you for installing {{ .Chart.Name }}!

Your release is named: {{ .Release.Name }}

To connect to Redis:
{{- if eq .Values.service.type "ClusterIP" }}
  kubectl port-forward svc/{{ include "redis-service.fullname" . }} {{ .Values.service.port }}:{{ .Values.service.port }}
  Then connect using: redis-cli -h localhost -p {{ .Values.service.port }}
{{- else if eq .Values.service.type "NodePort" }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "redis-service.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  redis-cli -h $NODE_IP -p $NODE_PORT
{{- else if eq .Values.service.type "LoadBalancer" }}
  kubectl get svc {{ include "redis-service.fullname" . }} -w
{{- end }}

Note: This Redis instance is running in-memory mode (no persistence).
      Data will be lost when the pod restarts.
